    }

     * å½•åƒè®¡åˆ’èµ„æºåŠ è½½å™¨

            recordPlansCacheService.set(result.getCameraIndexCode(), result.getType(), result);
        });
    }

    private void recordPlansCacheLoader(CacheOperations<String, RecordPlansRespDTO> cacheOperations) {
        Date startDate = DateUtil.date();
        int i = 1;
        while (true) {
            BaseResult<BasePageDTO<RecordPlansRespDTO>> basePageBaseResult = recordPlansRestServiceClient.recordPlansPageSearch(i++, DEFAULT_PAGE_SIZE);
            BasePageDTO<RecordPlansRespDTO> basePage = Optional.ofNullable(basePageBaseResult).map(BaseResult::getData).orElse(null);
            List<RecordPlansRespDTO> recordPlansRespDTOS = Optional.ofNullable(basePage).map(BasePageDTO::getList).orElse(Collections.emptyList());
            if (CollUtil.isEmpty(recordPlansRespDTOS)) {
                break;
            }
            recordPlansRespDTOS.forEach(recordPlansRespDTO -> {
                cacheOperations.set(null, recordPlansRespDTO);
            });
            Integer totalPage = Optional.ofNullable(basePage).map(BasePageDTO::getTotalPage).orElse(0);
            if (totalPage < i) {
                break;
            }

        long between = DateUtil.between(startDate, DateUtil.date(), DateUnit.MS);
        log.info("å½•åƒè®¡åˆ’èµ„æºåŠ è½½è€—æ—¶ï¼š {} æ¯«ç§’.", between);

    private void stationInfoCacheLoader(String userIndexCode) {
        resourceCacheLoader(userIndexCode, RESOURCE_TYPE_STATION, (indexCode, resultMap) -> {

                    .orElse(BeanUtil.mapToBean(resultMap, StationInfoDTO.class, false));
            Set<String> users = Optional.ofNullable(stationInfoDTO)
                    .map(StationInfoDTO::getUserIndexCodeSet).orElse(CollUtil.newHashSet(userIndexCode));
            users.add(userIndexCode);
            stationInfoDTO.setUserIndexCodeSet(users);
            stationCacheRedisService.set(indexCode, stationInfoDTO);
            stationCacheRedisService.set(stationInfoDTO.getCrossingIndexCode(), stationInfoDTO);
        });

        while (true) {
            Param4ResourceV2 param4ResourceV2 = Param4ResourceV2.builder()

            BaseResult<BasePage<Map<String, Object>>> basePageBaseResult = resourceRestServiceV2Client.searchResourcesByAuth(param4ResourceV2);
            List<Map<String, Object>> resultMapList = Optional.ofNullable(basePageBaseResult).map(BaseResult::getData).map(BasePage::getList).orElse(null);

                break;

                cacheOperations.set(resultMap.get(SearchConstants.Resource_KEY).toString(), resultMap);
            });

        log.info("[{}]èµ„æºåŠ è½½è€—æ—¶ï¼š {} æ¯«ç§’.", resourceType, between);

@Service
public class CrossCacheRedisService extends AbstractRedisService<String, CrossInfoDTO> {
    private static final String DEFAULT_KEY_PREFIX = "osurveillance:resourceCache:cross:";
    private static final String AT = "@";

    protected CrossCacheRedisService(@Qualifier("resourceCacheRedisTemplate") RedisTemplate<String, Object> redisTemplate) {
        super(redisTemplate);
    }

    @Override
    public String rebuildKey(String s) {
        if (StrUtil.startWith(s, DEFAULT_KEY_PREFIX)) {
            return s;
        }
        return StrUtil.concat(true, DEFAULT_KEY_PREFIX, s);
    }

    public String getCrossName(String indexCode) {
        return Optional.ofNullable(get(indexCode)).map(CrossInfoDTO::getName).orElse(StrUtil.EMPTY);
    }

    public List<String> getCrossTagList(String indexCode) {
        String tag = Optional.ofNullable(get(indexCode)).map(CrossInfoDTO::getTag).orElse(StrUtil.EMPTY);
        if (StrUtil.isBlank(tag)) {
            return CollUtil.newArrayList();
        }
        String[] split = StrUtil.split(tag, AT);
        return CollUtil.newArrayList(split);
    }

    public void deleteAll() {
        super.delete(super.keys(DEFAULT_KEY_PREFIX + Constants.ASTERISK));
    }
}

    public String getValue(DictTypeCodeEnum dictTypeCodeEnum, String key) {

        return Optional.ofNullable(dataDictDTO).map(DataDictDTO::getDataValue).orElse(Constants.UNKNOW);
