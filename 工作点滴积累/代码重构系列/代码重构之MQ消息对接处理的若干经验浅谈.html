@Slf4j

public class OverrunRecordVehicleTransactionDeal extends BaseTransactionHandle implements TransactionBatchHandle<ConsumerRecord<?, ?>, OverrunRecordWarnDTO> {
    private ExecutorService executorService;
    private List<IMessageConsumer<RealTimeRecord>> iMessageConsumerList;
    @Autowired
    private RealTimeRecordAdapter realTimeRecordAdapter;
    @Autowired
    private TransactionHandleHelp transactionHandleHelp;
    @Autowired
    private OverrunThresholdService overrunThresholdService;

    public OverrunRecordVehicleTransactionDeal(List<IMessageConsumer<RealTimeRecord>> iMessageConsumerList) {
        this.iMessageConsumerList = iMessageConsumerList;
        int size = iMessageConsumerList.size();
        this.executorService = new ThreadPoolExecutor(size, size + 1, 1L,
                TimeUnit.SECONDS, new LinkedBlockingQueue<>(size),
                Executors.defaultThreadFactory(), new ThreadPoolExecutor.AbortPolicy());
    }

    @Override
    public List<OverrunRecordWarnDTO> parseMessage(List<ConsumerRecord<?, ?>> message) {
        List<OverrunRecordDTO> overrunRecordDTOS = Optional.ofNullable(message)
                .map(transactionHandleHelp::consumerRecordGetMessage)
                .filter(transactionHandleHelp::checkBlank)
                .map(transactionHandleHelp::parseOverrunRecordDTOs)
                .orElse(CollUtil.newArrayList());
        List<OverrunRecordWarnDTO> overrunRecordWarnDTOList = overrunRecordDTOS.stream()
                .filter(transactionHandleHelp::checkEmpty)
                .filter(transactionHandleHelp::checkRequired)
                .filter(transactionHandleHelp::checkOverrunRecord)
                .map(transactionHandleHelp::toOverrunRecordWarnDTO)
                .filter(transactionHandleHelp::checkEmpty)
                .collect(Collectors.toList());
        return overrunRecordWarnDTOList;
    }

        List<RealTimeRecord> realTimeRecords = realTimeRecordAdapter.overrunRecordWarntoRealTimeRecords(overrunRecordWarnDTOList);
        List<RealTimeRecord> realTimeRecordsAfterThreshold = realTimeRecords.stream()
                .filter(ObjectUtil::isNotNull)
                .filter(overrunThresholdService::verifyData)
                .map(overrunThresholdService::isSevereOverrun).collect(Collectors.toList());
        if (CollUtil.isEmpty(realTimeRecordsAfterThreshold)) {

                iMessageConsumer.receive(realTimeRecordsAfterThreshold);
            });
        });
    }
}
